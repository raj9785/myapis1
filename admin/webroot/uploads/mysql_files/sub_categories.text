CREATE TABLE `sub_categories` (`id` int(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,`name` varchar(100) NOT NULL ,`category_id` int(11) NOT NULL ,`application_id` int(6) NOT NULL,`is_active` tinyint(1) NOT NULL DEFAULT '1',`created_on` datetime DEFAULT NULL,`created_by` int(11) NOT NULL, FOREIGN KEY (category_id)  REFERENCES categories(id) ON DELETE NO ACTION ON UPDATE CASCADE) ENGINE=InnoDB DEFAULT CHARSET=utf8;
